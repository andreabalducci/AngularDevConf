    <h3 ng-click="inc()">Counter is {{counter}}</h3>
</div>
<script>
    angular.module('demo', []); // define
    angular.module('demo') // reference
        .controller('demoCtrl', function($scope){
            $scope.counter = 0;
            $scope.inc = function(){
                $scope.counter++;
            }
        });
</script>
